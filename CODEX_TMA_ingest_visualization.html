```{python}
def plot_spatial_grid(adata, rows=10, cols=4, dpi=100, size=2.0):
    rows_labels = [chr(ord("A") + r) for r in range(rows)]
    cols_labels = [str(c) for c in range(1, cols + 1)]
    all_tma = adata[adata.obs["dataset"] == "TMA"]
    n_clusters = int(adata.obs["leiden"].nunique())
    cluster_colors = {str(i): cluster_palette[i] for i in range(n_clusters)}
    fig_w = cols * 2.5
    fig_h = rows * 2.8
    fig, axes = plt.subplots(rows, cols, figsize=(fig_w, fig_h), dpi=dpi)
    for row in range(rows):
        for col in range(cols):
            ax = axes[row, col]
            core = f"{rows_labels[row]}-{cols_labels[col]}"
            subset = all_tma[all_tma.obs["tma_label"] == core]
            if subset.n_obs == 0:
                ax.text(0.5, 0.5, f"{core}\n(empty)", ha="center", va="center",
                        fontsize=7, color="gray")
                ax.set_xlim(0, 1); ax.set_ylim(0, 1)
                ax.set_xticks([]); ax.set_yticks([])
                continue
            coords = subset.obsm["spatial_um"]
            leiden_vals = subset.obs["leiden"].values
            for cluster in subset.obs["leiden"].unique():
                mask = leiden_vals == cluster
                ax.scatter(coords[mask, 0], coords[mask, 1],
                           c=[cluster_colors[cluster]], s=size,
                           edgecolors="none", alpha=0.7)
            ax.set_title(f"{core} (n={subset.n_obs})", fontsize=7)
            ax.set_aspect("equal")
            ax.set_xticks([]); ax.set_yticks([])
    handles = [plt.Line2D([], [], marker="o", color=w, linestyle="None",
                          markersize=8, label=leiden_label.get(c, c))
               for c, w in cluster_colors.items()]
    fig.legend(handles=handles, title="Cell type", loc="upper center",
               ncol=4, fontsize=9, bbox_to_anchor=(0.5, 1.02))
    plt.tight_layout(rect=[0, 0, 1, 0.88])
    plt.show()
```

```{python}
def plot_umap_grid(adata, rows=10, cols=4, dpi=100, s=1):
    rows_labels = [chr(ord("A") + r) for r in range(rows)]
    cols_labels = [str(c) for c in range(1, cols + 1)]
    n_clusters = int(adata.obs["leiden"].nunique())
    cluster_colors = {str(i): cluster_palette[i] for i in range(n_clusters)}
    fig, axes = plt.subplots(rows, cols, figsize=(cols*1.8, rows*1.5), dpi=dpi)
    fig.suptitle("UMAP per TMA core", fontsize=12, y=0.98)
    for row in range(rows):
        for col in range(cols):
            ax = axes[row, col]
            core = f"{rows_labels[row]}-{cols_labels[col]}"
            subset = adata[adata.obs["tma_label"] == core]
            if subset.n_obs < 10:
                ax.text(0.5, 0.5, f"{core}\n(empty)", ha="center", va="center", fontsize=6, color="gray")
                ax.set_xlim(0, 1); ax.set_ylim(0, 1)
                ax.set_xticks([]); ax.set_yticks([])
                continue
            umap = subset.obsm["X_umap"]
            leiden_vals = subset.obs["leiden"].values
            for cluster in subset.obs["leiden"].unique():
                mask = leiden_vals == cluster
                ax.scatter(umap[mask, 0], umap[mask, 1], c=[cluster_colors[cluster]], s=s, edgecolors="none", alpha=0.6)
            ax.set_title(core, fontsize=6)
            ax.set_aspect("equal")
            ax.set_xticks([]); ax.set_yticks([])
    handles = [plt.Line2D([], [], marker="o", color=w, linestyle="None",
                          markersize=8, label=leiden_label.get(c, c))
               for c, w in cluster_colors.items()]
    fig.legend(handles=handles, title="Cell type", loc="upper center",
               ncol=4, fontsize=9, bbox_to_anchor=(0.5, 1.02))
    plt.tight_layout(rect=[0, 0, 1, 0.88])
    plt.show()
```

```{python}
def plot_tonsil_spatial(adata, sample_name, dpi=100, scale=0.0005, size=1.5):
    subset = adata[(adata.obs["dataset"] == "tonsil") & (adata.obs["sample"] == sample_name)]
    coords = subset.obsm["spatial"]
    x_range = coords[:, 0].max() - coords[:, 0].min()
    y_range = coords[:, 1].max() - coords[:, 1].min()
    fig_w = max(x_range * scale, 6)
    fig_h = max(y_range * scale, 6)
    fig, ax = plt.subplots(figsize=(fig_w, fig_h), dpi=dpi)
    sc.pl.embedding(
        subset,
        basis="spatial",
        color="leiden_label",
        title=sample_name,
        show=False,
        size=size,
        ax=ax,
    )
    plt.tight_layout()
    plt.show()
```
